세팅지 진행해주어야 할것

#초기 라라벨8세팅
composer install
php artisan key:generate
php artisan config:cache
php artisan migrate 
chmod -R 777 storage/
npm install
npm run dev

#제트스트림 로그인
composer require laravel/jetstream
php artisan jetstream:install livewire
php artisan jetstream:install inertia
npm install
npm run dev
php artisan migrate

#voyager 관리자 패키지 설치
composer require tcg/voyager
php artisan voyager:install  
(php artisan voyager:install --with-dummy) #더미데이터를 넣고 싶을때 해당것을 사용한다
php artisan voyager:admin  admin@email.com  #(관리자세팅) =>더미데이터로 진행했을때는 관리자 생성을 하지 않아도 된다. 더미데이터 admin@admin.com / password
php artisan voyager:admin admin@email.com --create
